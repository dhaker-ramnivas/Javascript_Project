            if(map.containsKey(tempStr))

                map.put(tempStr, map.get(tempStr) + 1);
            }
            else
            {
                map.put(tempStr,1);
            }
        }

        
        Iterator itr =map.entrySet().iterator();

            Map.Entry pair = (Map.Entry)itr.next();
            Integer frequency=(Integer)pair.getValue()

                nameOFPerson=(String)pair.getKey();
                count=frequency;

        }
